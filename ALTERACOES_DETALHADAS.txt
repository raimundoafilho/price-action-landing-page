│ function submitPassword() {              ← ⭐ VALIDAÇÃO DE SENHA           │
│   const input = document.getElementById('passwordInput');                  │
│   const pwd = input.value.trim();                                          │
│                                                                             │
│   if (pwd === CORRECT_PASSWORD) {                                          │
│     sessionStorage.setItem('authenticated', 'true');  ← SALVA EM MEMÓRIA  │
│     showLandingPage();                                                     │
│     input.value = '';                                                      │
│   } else {                                                                  │
│     const errorElement = document.getElementById('loginError');             │
│     errorElement.textContent = '❌ Senha incorreta. Tente novamente.';     │
│     input.value = '';                                                      │
│     input.focus();                                                          │
│   }                                                                         │
│ }                                                                           │
│                                                                             │
│ function logout() {                      ← ⭐ FUNÇÃO DE LOGOUT             │
│   sessionStorage.removeItem('authenticated');                              │
│   location.reload();                                                        │
│ }                                                                           │
│                                                                             │
│ function showLandingPage() {             ← ⭐ MOSTRA LANDING PAGE          │
│   const loginScreen = document.getElementById('login-screen');             │
│   const slideContainer = document.getElementById('slideContainer');        │
│                                                                             │
│   if (loginScreen) loginScreen.style.display = 'none';                    │
│   if (slideContainer) slideContainer.style.display = 'block';             │
│                                                                             │
│   initialize();                          ← CARREGA APP                    │
│ }                                                                           │
│                                                                             │
│ function checkAuthentication() {         ← ⭐ VERIFICA AUTENTICAÇÃO        │
│   const isAuthenticated = sessionStorage.getItem('authenticated') === 'true';
│   const loginScreen = document.getElementById('login-screen');             │
│   const slideContainer = document.getElementById('slideContainer');        │
│                                                                             │
│   if (isAuthenticated) {                                                    │
│     if (loginScreen) loginScreen.style.display = 'none';                  │
│     if (slideContainer) slideContainer.style.display = 'block';           │
│     showLandingPage();                                                     │
│   } else {                                                                  │
│     if (loginScreen) loginScreen.style.display = 'flex';                  │
│     if (slideContainer) slideContainer.style.display = 'none';            │
│                                                                             │
│     const passwordInput = document.getElementById('passwordInput');        │
│     if (passwordInput) {                                                    │
│       passwordInput.focus();                                                │
│                                                                             │
│       passwordInput.addEventListener('keypress', (e) => {                 │
│         if (e.key === 'Enter') {                                           │

│       });                                                                   │
│     }                                                                       │
│   }                                                                         │
│ }                                                                           │

   if (pwd === CORRECT_PASSWORD)
   → 'AndreTrader' ≠ 'andretrader'
   → 'AndreTrader' ≠ 'ANDRETRADER'

✅ Validação 2: Trim de espaços
   const pwd = input.value.trim();
   → Remove espaços do início e fim

   sessionStorage.getItem('authenticated') === 'true'

   if (e.key === 'Enter') { submitPassword(); }
   → Usuário pode pressionar Enter

   passwordInput.focus();
